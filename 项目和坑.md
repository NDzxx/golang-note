```golang
package main  
import ( 
"container/list" 
"fmt" ) 
func main() {
 l := list.New() 
l.PushBack(1) 
l.PushBack(2) 
l.PushBack(3) 
l.PushBack(4)
fmt.Println("Before Removing...") 
//遍历list，删除元素错误做法
 /*for e := l.Front(); e != nil; e = e.Next() { 
fmt.Println("removing", e.Value)
 l.Remove(e) 
}*/ 
var n *list.Element 
for e := l.Front(); e != nil; e = n {
 fmt.Println("removing", e.Value) 
n = e.Next() 
l.Remove(e)
 } 
fmt.Println("After Removing...")
for e := l.Front(); e != nil; e = e.Next() 
{ 
fmt.Println(e.Value) 
} 
}
```
